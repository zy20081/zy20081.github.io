ASP.net web自带安全机制

Web.config配置

<configuration>
  <location path="defalut.aspx">
    <system.web>
      <authorization>
        <allow users="?"/>
      </authorization>
    </system.web>
  </location>
  <location path="Cinema3D">
    <system.web>
      <authorization>
        <deny users="2D"/>
      </authorization>
    </system.web>
  </location>
  <location path="student">
    <system.web>
      <authorization>
        <allow  users="?" />
      </authorization>
    </system.web>
  </location>
    <system.web>
      <authentication mode="Forms">
        <forms name="test" loginUrl="index.aspx" defaultUrl="defalut.aspx" cookieless="AutoDetect"></forms>
      </authentication>
      <authorization>
        <deny users="?"/>
      </authorization>
        <compilation debug="true" targetFramework="4.0" />
    </system.web>

</configuration>
