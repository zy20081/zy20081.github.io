        /// <summary>
        /// 序列化存进2进制流
        /// </summary>
        /// <param name="fileName">路径</param>
        private void LoadDate(string fileName)
        {
            #region 序列化
            FileStream fs = new FileStream(fileName, FileMode.OpenOrCreate);
            BinaryFormatter bf = new BinaryFormatter();//创建序列化
            bf.Serialize(fs, m_RM);//保存
            bf = null;
            fs.Close();
            fs.Dispose(); 
            #endregion
        }
